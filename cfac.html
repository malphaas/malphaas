Smallish C program to compute 100!

My solution for Project Euler problem 20, base used is 10^9, simple schoolbook multiplication algorithm. ~281 bytes. Certainly not the smallest possible. 

#include <stdio.h>
#include <stdint.h>
const uint32_t B=1e9;
#define N 18
int main(){uint32_t r[N]={479001600},a,k;uint64_t d;int i;for(a=13;a<=100;a++)for(i=0,k=0;i<N;r[i++]=d%B,k=d/B)d=(uint64_t)r[i]*a+k;for(i=0,a=0;i<N;i++)for(k=r[i];k;k/=10)a+=k%10;printf("%d\n",a);return 0;}
